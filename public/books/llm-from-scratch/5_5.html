Temperature is your creativity dial: lower values produce focused, predictable outputs while higher values encourage exploration. The formula is elegantly simple‚Äîjust divide logits by T before softmax‚Äîbut the effect on generation quality is profound.
PyTorch Implementation
Core Function: softmax_with_temperature
def softmax_with_temperature(logits, temperature):
    """
    Applies softmax to the logits with the specified temperature.
    
    Parameters:
    - logits (torch.Tensor): Raw output scores from the model
    - temperature (float): Temperature value to scale the logits
    
    Returns:
    - torch.Tensor: Probability distribution after softmax
    """
    # Scale the logits by dividing by the temperature
    scaled_logits = logits / temperature
    # Apply softmax to convert scaled logits to probabilities
    probabilities = torch.softmax(scaled_logits, dim=-1)
    return probabilities
Example Usage
# Example vocabulary (token_id: word)
vocab = {
    "closer": 0, "every": 1, "effort": 2,
    "forward": 3, "inches": 4, "moves": 5,
    "pizza": 6, "toward": 7, "you": 8,
}
inverse_vocab = {v: k for k, v in vocab.items()}

# Logits for next token after "Every effort moves you"
next_token_logits = torch.tensor(
    [4.51, 0.89, -1.90, 6.75, 1.63, -1.62, -1.89, 6.28, 1.79]
)

# Compare different temperatures
temperatures = [0.1, 1, 5]
scaled_probas = [softmax_with_temperature(next_token_logits, T) 
                 for T in temperatures]

for T, probs in zip(temperatures, scaled_probas):
    print(f"\nTemperature = {T}")
    for i, prob in enumerate(probs):
        print(f"  {inverse_vocab[i]}: {prob.item():.4f}")
Greedy vs. Probabilistic Decoding
# GREEDY DECODING (deterministic)
# Always selects the highest probability token
probas = torch.softmax(next_token_logits, dim=0)
next_token_id = torch.argmax(probas).item()
print(inverse_vocab[next_token_id])  # "forward"

# PROBABILISTIC SAMPLING (with temperature)
# Samples from the probability distribution
torch.manual_seed(123)
next_token_id = torch.multinomial(probas, num_samples=1).item()
print(inverse_vocab[next_token_id])  # Could be "toward" or others
